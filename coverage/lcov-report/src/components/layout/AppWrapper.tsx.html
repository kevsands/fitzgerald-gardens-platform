'use client';
 
import React, { ReactNode } from "react";
import { useAuth } from "@/context/AuthContext";
import Link from "next/link";
import { ErrorBoundary } from 'react-error-boundary';
import AmplifyProvider from "@/components/AmplifyProvider";
 
// Define AppHeader component with proper typing
const AppHeader = ({ isAuthenticated, user, signOut }: { 
  isAuthenticated: boolean, 
  user: any, // Consider updating to a more specific type
  signOut: () => void 
}) => (
  <header className="bg-gray-800 text-white p-4 shadow-md sticky top-0 z-50">
    <div className="container mx-auto flex justify-between items-center">
      <Link href="/" className="text-xl font-bold hover:opacity-80 transition-opacity">Prop.ie</Link>
      <nav className="space-x-4 flex items-center">
        <Link href="/properties" className="hover:text-gray-300 transition-colors">Properties</Link>
        <Link href="/developments" className="hover:text-gray-300 transition-colors">Developments</Link>
        {isAuthenticated ? (
          <>
            <span className="hidden sm:inline">Welcome, {user?.username || user?.email || "User"}</span>
            <button 
              onClick={signOut} 
              className="hover:text-gray-300 transition-colors px-2 py-1 rounded focus:outline-none focus:ring-2 focus:ring-white"
              aria-label="Sign Out"
            >
              Sign Out
            </button>
          </>
        ) : (
          <>
            <Link href="/login" className="hover:text-gray-300 transition-colors">Login</Link>
            <Link 
              href="/register" 
              className="bg-white text-gray-800 px-3 py-1 rounded hover:bg-gray-200 transition-colors focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-offset-gray-800 focus:ring-white"
            >
              Register
            </Link>
          </>
        )}
      </nav>
    </div>
  </header>
);
 
const AppFooter = () => (
  <footer className="bg-gray-200 text-gray-700 p-4 mt-auto">
    <div className="container mx-auto text-center text-sm">
      &copy; {new Date().getFullYear()} Prop.ie. All rights reserved.
    </div>
  </footer>
);
 
interface AppWrapperProps {
  children: ReactNode;
}
 
/**
 * AppWrapper Component
 * Provides the main application layout structure (header, main content, footer).
 * Includes ErrorBoundary to catch rendering errors in children.
 * Adapts based on authentication state from AuthContext.
 */
const AppWrapper: React.FC<AppWrapperProps> = ({ children }) => {
  const { user, isAuthenticated, signOut } = useAuth();
 
  return (
    <AmplifyProvider>
      <div className="flex flex-col min-h-screen bg-gray-100">
        {/* Render Header, passing auth state */}
        <AppHeader isAuthenticated={isAuthenticated} user={user} signOut={signOut} />
 
        {/* Main content area wrapped in ErrorBoundary */}
        <main className="flex-grow container mx-auto px-4 py-6">
          <ErrorBoundary fallback={
            <div className="p-4 bg-red-50 border border-red-200 rounded-md">
              <h2 className="text-lg font-semibold text-red-800">Something went wrong</h2>
              <p className="text-red-600">
                Please try refreshing the page or contact support if the issue persists.
              </p>
            </div>
          }>
            {/* Render the page content passed as children */}
            {children}
          </ErrorBoundary>
        </main>
 
        {/* Render Footer */}
        <AppFooter />
      </div>
    </AmplifyProvider>
  );
};
 
export default AppWrapper;
